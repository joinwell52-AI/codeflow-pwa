## §四 验收步骤（建议顺序）

```powershell
# 0. 工作目录与版本核对
cd D:\Bridgeflow\codeflow-shell
git log --oneline -3       # 期望看到 fix(s6-v0.2-sprint0-mt1-hotfix) commit
git tag --contains HEAD    # 期望含 v0.2.0-beta.1（或近期）
node -e "console.log(require('./package.json').version)"
# 期望: 0.2.0-beta.1

# 1. 环境核对（不读 .env 内容）
Test-Path codeflow-shell\.env       # 期望 True
(Get-Item codeflow-shell\.env).Length    # 期望 ~84 bytes（同 QA-009）
git status --short codeflow-shell\.env   # 期望空

# 2. 启动 + 看 banner
npm start
# 期望 banner 关键字：
#   "Cursor SDK     : live (CursorSdkAdapter"
#   "listScope=\"local\""
#   "Skills loaded  : 3"
#   "Status         : running. Drop TASK-*-XXX-to-AGENT.md to inbox."
# 注意：如果出现 WARNING 块（live + local + no model），记录但不算 fail

# 3. drop sample task
# 在另一窗口跑：
Copy-Item codeflow-shell\sample-task.md codeflow-shell\inbox\TASK-20260510-999-PM-to-DEV.md

# 等 60-180 秒（真 LLM 调用，比 fake 慢）
# 观察 main shell stdout：
#   [TaskDispatcher] received TASK-20260510-999...
#   [SessionManager] startSession session_id=...
#   [SDK] (no error 'Local SDK agents require an explicit model')
#   [ReviewEngine] review created review_id=REVIEW-20260510-999-...
#   [ReviewEngine] decision=<one of: approved/rejected/needs_changes/abstained>

# 4. 看 review 文件（不贴 body 进 REPORT）
Get-ChildItem reviews\REVIEW-20260510-999-*.md | Select-Object -First 1
# Read frontmatter only:
Get-Content (Get-ChildItem reviews\REVIEW-20260510-999-*.md)[0] -TotalCount 30
# 截 frontmatter 到 REPORT：decision / session_id / review_id / task_ref / verdict_source

# 5. 看 transcript（不贴 body 进 REPORT）
$tx = Get-ChildItem data\transcripts\*.jsonl | Sort-Object LastWriteTime -Descending | Select-Object -First 1
$tx.Length     # 期望 > 100 bytes
Get-Content $tx | Select-Object -First 1 | ConvertFrom-Json | Select-Object event_type, sdk_agent_id
# 只贴 event_type + sdk_agent_id 到 REPORT，不贴 content 字段

# 6. Ctrl+C 优雅停机
# 观察 stdout: "Shutting down..." + "Runtime stopped."
```
