### 2. `src/main.ts` — 入口逻辑（参考 design doc §11.3 Layer 1）

```typescript
async function main() {
  const dataDir = process.env.CODEFLOW_DATA_DIR ?? path.join(os.homedir(), ".codeflow");
  const runtime = await Runtime.create({
    persistDir: dataDir,
    inboxDir: path.join(dataDir, "inbox"),
    transcriptsDir: path.join(dataDir, "transcripts"),
    reviewsDir: path.join(dataDir, "reviews"),
    skillsDir: path.join(dataDir, "skills"),
    sdk: makeRealCursorSdkAdapter() ?? makeFakeCursorSdkAdapter(),
  });

  await runtime.bootstrap();
  await runtime.start();

  console.log("===========================================================");
  console.log("CodeFlow v0.1.0-rc.1 — internal preview");
  console.log("===========================================================");
  console.log(`Data dir       : ${dataDir}`);
  console.log(`Inbox          : ${runtime.inboxDir}`);
  console.log(`Status         : running. Drop TASK-*-XXX-to-AGENT.md to inbox.`);
  console.log(`Stop           : Ctrl+C`);
  console.log("===========================================================");

  process.on("SIGINT", async () => {
    console.log("\n[shell] received SIGINT, stopping runtime...");
    await runtime.stop();
    process.exit(0);
  });
}

main().catch((err) => {
  console.error("[shell] fatal:", err);
  process.exit(1);
});
```

### 4. `sea-config.json` + `pack.cmd`（Windows MVP）

```jsonc
// sea-config.json
{
  "main": "dist/main.js",
  "output": "dist/sea-prep.blob",
  "disableExperimentalSEAWarning": true
}
```

```bat
:: pack.cmd
@echo off
setlocal
echo === CodeFlow-Shell pack (Windows) ===
echo step 1: tsc compile
call npx tsc --project tsconfig.json
if errorlevel 1 exit /b 1

echo step 2: bundle dependencies (esbuild single file)
call npx esbuild dist/main.js --bundle --platform=node --target=node22 --outfile=dist/main.bundle.js
if errorlevel 1 exit /b 1

echo step 3: tweak sea-config.json to use bundle
:: sea-config.json points to dist/main.bundle.js
node --experimental-sea-config sea-config.json
if errorlevel 1 exit /b 1

echo step 4: copy node binary
copy "%ProgramFiles%\nodejs\node.exe" "dist\CodeFlow-v0.1.0-rc.1.exe"
if errorlevel 1 (
  echo Warning: copy failed; please manually copy node.exe
  exit /b 1
)

echo step 5: inject blob into exe
:: postject is the official Node SEA blob injection tool
call npx postject "dist\CodeFlow-v0.1.0-rc.1.exe" NODE_SEA_BLOB dist\sea-prep.blob ^
  --sentinel-fuse NODE_SEA_FUSE_fce680ab2cc467b6e072b8b5df1996b2

echo === Build complete: dist\CodeFlow-v0.1.0-rc.1.exe ===
endlocal
```

DEV 自测 1 次产出 EXE 即可（不必跑过 ADMIN 的双击 — 这是 §五验收 1 项）。如 Node SEA 当前环境上跑不动（已知风险：design doc §11.7 line 2678），可以**降级到 fallback：直接 npm script 启动**（`node dist/main.bundle.js`），EXE 留空 — 在 README 写"v0.1 internal RC 暂用 npm script，v0.2 sprint 0 后再补 EXE"。这是 PM 接受的 fallback。
